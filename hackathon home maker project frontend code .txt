function renderHome() {

  // Ring
  const circumference = 251.2;
  const offset = circumference - (pct / 100) * circumference;
  const ringEl = document.getElementById('ring-progress');

  if (addAmtStr==='0') addAmtStr=d; else addAmtStr+=d;
  if (addAmtStr.length>7) addAmtStr=addAmtStr.slice(0,7);
  document.getElementById('add-amount').textContent = parseInt(addAmtStr).toLocaleString();
}

function numDel() {
  addAmtStr = addAmtStr.slice(0,-1)||'0';
  document.getElementById('add-amount').textContent = parseInt(addAmtStr).toLocaleString();
}

function addExpense() {
  const amt = parseInt(addAmtStr);
  if (amt<=0)       { showToast('⚠️ Please enter an amount!'); return; }
  if (!selectedCat) { showToast('⚠️ Please select a category!'); return; }
  const cat = CATS.find(c=>c.id===selectedCat);
  const now = new Date();
  const timeStr = now.toLocaleTimeString('en-IN',{hour:'2-digit',minute:'2-digit'});
  state.transactions.unshift({
    id: Date.now(), cat: selectedCat,
    name: cat.name, amount: amt,
    date: 'Today, '+timeStr
  });
  showToast('✅ ₹'+amt.toLocaleString()+' added to '+cat.name+'!');
  addAmtStr='0'; selectedCat=null;
  setTimeout(()=>goTab('home'), 800);
}

function renderChart(period) {
  const labels = period==='week'
    ? ['Mon','Tue','Wed','Thu','Fri','Sat','Sun']
    : ['Week 1','Week 2','Week 3','Week 4'];
  const data = period==='week'
    ? [450,820,320,960,1200,680,340]
    : [4200,3800,5100,2900];
  const max = Math.max(...data);
  const todayIdx = period==='week' ? 6 : 3;

  document.getElementById('chart-bars').innerHTML = data.map((v,i)=>`
    <div class="bar-col">
      <div class="bar-bar" style="height:${Math.round(v/max*110)+10}px;background:${i===todayIdx?'var(--saffron)':i===todayIdx-1?'var(--saffron-mid)':'var(--border)'}"></div>
    </div>
  `).join('');

  document.getElementById('chart-labels').innerHTML = labels.map((l,i)=>`
    <span style="font-size:10px;color:${i===todayIdx?'var(--saffron)':'var(--text3)'};flex:1;text-align:center">${l}</span>
  `).join('');
}
